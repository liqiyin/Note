```java
//这行hin关键
@AutoService(Processor.class)
public class MyProcessor extends AbstractProcessor {
    //用来生成.java文件
    private Filer mFiler;
    //用来输出日志 当做log用
    private Messager mMessager;
    //分析文件的工具
    private Elements mElementUtils;

    @Override
    public synchronized void init(ProcessingEnvironment processingEnvironment) {
        super.init(processingEnvironment);
        mFiler = processingEnvironment.getFiler();
        mMessager = processingEnvironment.getMessager();
        mElementUtils = processingEnvironment.getElementUtils();
    }

    @Override
    public Set<String> getSupportedAnnotationTypes() {
        Set<String> annotations = new LinkedHashSet<>();
        annotations.add(MyAnnotation.class.getCanonicalName());
        return annotations;
    }

    @Override
    public SourceVersion getSupportedSourceVersion() {
        return SourceVersion.latestSupported();
    }

    @Override
    public boolean process(Set<? extends TypeElement> set, RoundEnvironment roundEnvironment) {
        Set<? extends Element> myAnnotationsElements = roundEnvironment.getElementsAnnotatedWith(MyAnnotation.class);
        for (Element element : myAnnotationsElements) {
            //1.包名
            PackageElement packageElement = mElementUtils.getPackageOf(element);
            String pkgName = packageElement.getQualifiedName().toString();
            mMessager.printMessage(Diagnostic.Kind.NOTE, "packageName: " + pkgName);

            //2.获取包装类类名
            TypeElement enclosingElement = (TypeElement) element.getEnclosingElement();
            String enclosingName = enclosingElement.getQualifiedName().toString();
            mMessager.printMessage(Diagnostic.Kind.NOTE, "enclosingClass:" + enclosingName);

            VariableElement variableElement = (VariableElement) element;
            //3.获取注解的成员变量名
            String varName = variableElement.getSimpleName().toString();
            //3.获取注解的成员变量类型
            String varClsType = variableElement.asType().toString();

            //4.获取注解类的元数据
            MyAnnotation myAnnotation = element.getAnnotation(MyAnnotation.class);
            int value = myAnnotation.value();
            mMessager.printMessage(Diagnostic.Kind.NOTE, varClsType + ":" + varName + "=====>" + value);

            //4.生成文件
            createFile(enclosingElement, varClsType, varName, value);
            return true;
        }
        return false;
    }

    private void createFile(TypeElement enclosingElement, String varClassType, String varName, int value) {
        String pkgName = mElementUtils.getPackageOf(enclosingElement).getQualifiedName().toString();
        try {
            JavaFileObject jfo = mFiler.createSourceFile(pkgName + ".MyAnnotation_Generate");
            Writer writer = jfo.openWriter();
            writer.write(brewCode(pkgName, varClassType, varName, value));
            writer.flush();
            writer.close();
        } catch (Exception e) {
        }
    }

    private String brewCode(String pkName, String bindViewFiledClassType, String bindViewFiledName, int id) {
        StringBuilder builder = new StringBuilder();
        builder.append("package " + pkName + ";\n\n");
        builder.append("//Auto generated by apt,do not modify!!\n\n");
        builder.append("public class MyAnnotation_Generate { \n\n");
        builder.append("public static void main(String[] args){ \n");
        String info = String.format("%s %s = %d", bindViewFiledClassType, bindViewFiledName, id);
        builder.append("System.out.println(\"" + info + "\");\n");
        builder.append("}\n");
        builder.append("}");
        return builder.toString();
    }
}
```
